简单的排序实现#

快速排序#

快排 分治思想 复杂度 [nlogn, n^2] 不稳定 –x随机取
排序区间为 [l, r] 时，长度小于 1，直接退出，否则选一个数字 x 作为比较元素
将大于 x 的放右边，小于 x 的放左边，等于 x 的随意放
确定 x 的位置后，对两侧继续递归

 1void quicksort(int l, int r) {
 2    if (l >= r) return; // 长度小于 1，直接退出  
 3    swap(a[l], a[l + rand() % (r - l + 1)]); // 保证 x 随机取  
 4    int x = a[l];  
 5    int i = l, j = r;  
 6    while (i < j) {  
 7        while (i < j && a[j] > x) // 不能写成 a[j] >= x  
 8            j--;  
 9        if (i < j)   
10            a[i++] = a[j];  
11        while (i < j && a[i] < x) // 不能写成 a[i] <= x  
12            i++;  
13        if (i < j)    
14            a[j--] = a[i];   
15    }   
16    a[i] = x;  
17    quicksort(l, i - 1); // 不能递归 i  
18    quicksort(i + 1, r);      
19}

 1void Quicksort(int l, int r){   
 2    if(l>=r)return;   
 3    int b[100001],c[100001];  
 4    int x=a[l+rand()&(r-l+1)];  
 5    int l1=0,l2=0,y=0;  
 6    for(int i=l;i<=r;i++){  
 7        if(a[i]<x)  
 8            b[++l1]=a[i];  
 9        else   
10            if(a[i]>x)  
11                c[++l2]=a[i];  
12            else ++y;  
13    }  
14    for(int i=1;i<=l1;i++)
15        a[l+i-1]=b[i];  
16    for(int i=1;i<=y;i++)  
17        a[l+l1+i-1]=x;  
18    for(int i=1;i<=l2;i++)  
19        a[l+l1+y+i-1]=c[i];  
20    Quicksort(l,l+l1-1);  
21    Quicksort(l+l1+y,r);  
22}

归并排序#

归并排序 分治 复杂度 nlogn 且稳定 要排序 [l, r]，长度为1直接退出，否则分为 [l, m],[m+1, r]; 递归两个子区间进行归并排序 将排序好的子区间合并

 1void mergesort(int l,int r){
 2    if(l==r)return;
 3    int m=(l+r)/2;
 4    mergesort(l,m);
 5    mergesort(m+1,r);
 6    int p1=l,p2=m+1,tot=0;
 7    while(p1<=m&&p2<=r){
 8        if(a[p1]<=a[p2])
 9            c[++tot]==a[p1++];
10        else
11            c[++tot]=a[p2++];
12    }
13    while(p1<=m)
14        c[++tot]=a[p1++];
15    while(p2<=m)
16        c[++tot]=a[p2++];
17    for(int i=1;i<=tot;i++)
18        a[i+l-1]=c[i];
19}

计数排序#

计数排序 适合值域范围较小 复杂度 n+k 稳定 统计每个数字出现了几次 统计完出现次数，求前缀和，可知道每个数字在拍完序的位置的范围 保证稳定性，倒着确定原本每个位置上的数字最后排在低级位

 1void countingsort(){
 2    memset(c,0,sizeof(c));
 3    for(int i=1;i<=n;i++)
 4        ++c[a[i]];
 5    for(int i=1;i<=m;i++)
 6        for(int j=1;j<=c[i];j++)
 7            printf("%d",i);
 8    printf("\n");
 9
10    for(int i=2;i<=m;i++)
11        c[i]+=c[i-1];
12    for(int i=n;i;i--)
13        r[i]=c[a[i]]--;
14    for(int i=1;i<=n;i++)
15        printf("%d",r[i]);
16    printf("\n");
17}

基数排序#

基数排序 复杂度 nk 拆分成 m 个关键字 从后往前 依次对 m 个关键字进行排序 每次排序会使用上一次排序的结果 一般使用计数排序来完成每次的排序 例如对三位数排序 先排个位 再排十位 再排百位 经常被用于字符串的排序 后缀数组的核心就是基数排序

 1void countingsort(){
 2    memset(c,0,sizeof(c));
 3    for(int i=1;i<=n;i++)
 4        ++c[v[i]];
 5    for(int i=1;i<=9;i++)
 6        c[i]+=c[i-1];
 7    for(int i=n;i;i--)
 8        r[sa[i]]=c[v[sa[i]]]--;
 9    for(int i=1;i<=n;i++)
10        sa[r[i]]=i;
11}
12
13void radixsort(){
14    for(int i=1;i<=n;i++)
15        sa[i]=i;
16    int x=1;
17    for(int i=1;i<=m;i++,x*=10){
18        for(int j=1;j<=n;j++){
19            v[j]=a[j]/x%10;
20        }
21        countingsort();
22    }
23}
