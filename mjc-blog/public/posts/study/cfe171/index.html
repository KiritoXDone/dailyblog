#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=3e5+7, mod=1e9+7;

int n,q,l,r;
ll a[N],s[N],u[N],t[N],ts[N];
ll x,y,z,v;

ll bs(int x){
    ll l=1,r=n,ans=1;
    while(l<=r){
        ll mid=(l+r)>>1;
        if((n*mid-mid*(mid-1)/2)<x)l=mid+1;
        else r=mid-1,ans=mid;
    }
    return ans;
}

void solve(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        s[i]=s[i-1]+a[i];
        u[i]=u[i-1]+s[i];
    }
    for(int i=n;i>0;i--){
        t[i]=t[i+1]+(n-i+1)*a[i];
    }
    for(int i=1;i<=n;i++){
        ts[i]=ts[i-1]+t[i];
    }
    cin>>q;
    while(q--){
        cin>>l>>r;
        x=bs(l),y=bs(r);
        ll ans=ts[y]-ts[x-1];
        z=n-(x*n-x*(x-1)/2-l);
        v=n-(y*n-y*(y-1)/2-r);
        ans-=u[z-1]-u[x-1]-(z-x)*s[x-1];
        ans-=u[n]-u[v]-s[y-1]*(n-v);
        cout<<ans<<endl;
    }
}
