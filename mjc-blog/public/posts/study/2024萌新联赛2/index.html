矩阵计算板

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4int n,m;
 5int a[6][6];
 6
 7int cal(int n, int a[6][6]) {
 8    if (n == 1)
 9        return a[0][0];
10    int res = 0;
11    int sub[6][6];
12    for (int x = 0; x < n; x++) {
13        int subi = 0;
14        for (int i = 1; i < n; i++) {
15            int subj = 0;
16            for (int j = 0; j < n; j++) {
17                if (j == x) continue;
18                sub[subi][subj] = a[i][j];
19                subj++;
20            }
21            subi++;
22        }
23        res += (x % 2 == 0 ? 1 : -1) * a[0][x] * cal(n - 1, sub);
24    }
25    return res;
26}
27
28int main() {
29    cin >> n >> m;
30    for (int i = 0; i < n; i++)
31        for (int j = 0; j < m; j++)
32            cin >> a[i][j];
33    int ans = INT_MAX;
34	int u=min(n,m);
35	for(int i=1;i<=u;i++){
36		ans=min(ans,cal(i,a));
37	}
38    cout << ans << endl;
39    return 0;
40}

 1#include <bits/stdc++.h>
 2using namespace std;
 3const int N = 1e7+10;
 4typedef long long ll;
 5
 6int a[N],sum[N],ans[N<<1];
 7string A,B;
 8
 9void solve(){
10	cin>>A>>B;
11	int la=A.length(),lb=B.length(),b=B[0]-'0';
12	ll last=0;
13	a[la+1]=0;
14	for(int i=1;i<=la;i++){
15		a[i]=(A[la-i]-'0')*b+last;
16		last=a[i]/10;
17		a[i]*=10;
18		sum[i]=sum[i-1]+a[i];
19	}
20	sum[la+1]=sum[la]+last;
21	last=0;
22	for(int i=1;i<=la+lb;i++){
23		int l=min(i,la+1),r=max(0,i-lb);
24		ans[i]=sum[l]-sum[r]+last;
25		last=ans[i]/10;
26		ans[i]%=10;
27	}
28    bool flag=0;
29    for(int i=la+lb;i>=1;--i) {
30        if(!ans[i]) {
31            if(flag)  printf("%d",ans[i]);
32        } else {
33            flag=1;
34            printf("%d",ans[i]);
35        }
36    }
37    if(!flag)  printf("0");
38    printf("\n");
39}
40
41int main(){
42	int t;
43	cin>>t;
44	while(t--)solve();
45	return 0;
46}

E “好"字符#

观察得到 a，b 同一字符所处位置相邻差值构成的一个循环同构如果相同，就符合题意
那么就对 26 个字符各跑一次 找到位置 存入字符串 找到该串的最小表示 比较即可
需要注意将原 a，b 再复制本身一遍 因为这样才能保证找的最小表示没有缺少

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4int kmp(int nxt[], string a) {
 5    int n = a.size();
 6    nxt[0] = 0;
 7    int j = 0;
 8    for (int i = 1; i < n; i++) {
 9        while (j && a[i] != a[j])
10            j = nxt[j - 1];
11        if (a[i] == a[j])
12            j++;
13        nxt[i] = j;
14    }
15    return n - nxt[n - 1];
16}
17
18string getmin(string s, int n) {
19    s = s + s;
20    int i = 0, j = 1;
21    while (j < n) {
22        int k = 0;
23        while (k < n && s[i + k] == s[j + k]) ++k;
24        if (s[i + k] > s[j + k]) i += k + 1;
25        else j += k + 1;
26        if (i == j) j++;
27        if (i > j) swap(i, j);
28    }
29    string t = s.substr(i, n);
30    return t;
31}
32
33int main() {
34    int n;
35    string a, b;
36    cin >> n >> a >> b;
37    a = a + a;
38    b = b + b;
39    map<char, vector<int>> va, vb;
40    map<char, string> sa, sb;
41    for (int i = 0; i < n * 2; i++) {
42        va[a[i]].push_back(i);
43        vb[b[i]].push_back(i);
44    }
45    int cnt = 0, nxt[2000010];
46    for (char x = 'a'; x <= 'z'; x++) {
47        if (va[x].size() != vb[x].size() || va[x].size() == 0 || vb[x].size() == 0) continue;
48        for (int i = 1; i < va[x].size(); i++) {
49            sa[x] += to_string(va[x][i] - va[x][i - 1]);
50        }
51        for (int i = 1; i < vb[x].size(); i++) {
52            sb[x] += to_string(vb[x][i] - vb[x][i - 1]);
53        }
54        if (getmin(sa[x], kmp(nxt, sa[x])) == getmin(sb[x], kmp(nxt, sb[x]))) cnt++;
55    }
56    cout << cnt << endl;
57    return 0;
58}

G lxy 的通风报信#

因为数据不大，直接 bfs 跑每个点对其他点的距离，再求最小生成树即可

 1#include <bits/stdc++.h>
 2using namespace std;
 3const int N = 1010;
 4
 5struct Node{
 6    int x,y,id;
 7}b[N];
 8
 9int n,m,ans,dx[4]={1,0,-1,0},dy[4]={0,1,0,-1},tot,dis[N][N],a[N][N];
10bool vis[N];
11deque<Node>q;
12
13void bfs(int id){
14    q.clear();
15    q.push_back(Node{b[id].x,b[id].y,0});
16    dis[b[id].x][b[id].y]=0;
17    while(!q.empty()){
18        Node now=q.front();
19        q.pop_front();
20        for(int i=0;i<4;i++){
21            int x=now.x+dx[i],y=now.y+dy[i];
22            if(x<1||y<1||x>n||y>m||a[x][y]==-1)continue;
23            if(dis[x][y]>now.id+1){
24                dis[x][y]=now.id+1;
25                q.push_back(Node{x,y,dis[x][y]});
26            }
27        }
28    }
29}
30
31void solve(){
32    cin>>n>>m;
33    for(int i=1;i<=n;i++){
34        for(int j=1;j<=m;j++){
35            dis[i][j]=1e9;
36            char x;
37            cin>>x;
38            if(x=='.')a[i][j]=0;
39            else if(x=='#')a[i][j]=-1;
40            else {
41                a[i][j]=++tot;
42                b[tot]=Node{i,j,tot};   
43            }
44        }
45    }
46    dis[b[1].x][b[1].y]=0;
47    for(int i=1;i<=tot;i++){
48        int mi=1e9,id=0;
49        for(int j=1;j<=tot;++j){
50            if(vis[j])continue;
51            if(mi>dis[b[j].x][b[j].y])mi=dis[b[j].x][b[j].y],id=j;
52        }
53        if(!id){
54            puts("No");
55            return;
56        }
57        vis[id]=1;
58        ans+=mi;
59        if(i!=tot)bfs(id);
60    }
61    cout<<ans<<endl;
62}
63
64int main(){
65    int t=1;
66//     cin>>t;
67    while(t--)solve();
68    return 0;
69}
