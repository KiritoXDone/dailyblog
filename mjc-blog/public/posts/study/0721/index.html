#include <bits/stdc++.h>
using namespace std;
const int N = 2e7+10;
typedef long long ll;

string s,p;
ll z[N*2],n,m;

void exkmp(string s,string p){
    n=s.size();
    m=p.size();
    p=p+"#"+s;
    ll L=0,R=-1;
    z[0]=m;
    for(int i=1;i<n+m+1;i++){
        if(i>R)
            z[i]=0;
        else{
            ll k=i-L;
            z[i]=min(z[k],R-i);
        }
        while(i+z[i]<n+m+1&&p[z[i]]==p[i+z[i]])
            ++z[i];
        if(i+z[i]-1>R)
            L=i,R=i+z[i]-1;
    }
    ll ans=0;
    for(int i=0;i<m;i++)ans^=(1LL*(i+1)*(z[i]+1));
    cout<<ans<<"\n";
    ans=0;
    for(int i=0;i<n;i++)ans^=(1LL*(i+1)*(z[i+m+1]+1));
    cout<<ans<<"\n";
}

int main(){
    cin>>s>>p;
    exkmp(s,p);
    return 0;
}
