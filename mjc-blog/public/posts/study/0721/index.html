子串查询#

输入两个字符串 s, p,查询 p 是否在 s 中出现,若出现输出出现位置,否则输出 -1

简化版

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4string s,p;
 5int nxt[100001],f[100001],n,m;
 6
 7void kmp(){
 8    n=s.size();
 9    m=p.size();
10    int j=0;
11    nxt[0]=0;
12    p=p+"#"+s;
13    for(int i=1;i<m+n+1;i++){
14        while(j&&p[i]!=p[j])
15            j=nxt[j-1];
16        if(p[i]==p[j])
17            j++;
18        nxt[i]=j;
19    }
20    for(int i=m+1;i<n+m+1;i++){
21        if(nxt[i]==m)cout<<i-2*m+1<<endl;
22    }
23}
24
25int main(){
26    cin>>s>>p;
27    kmp();
28    return 0;
29}

Secret word#

给出字符串 s，寻找最长字符串 p，p 满足（是 s 的子串，翻转后是 s 的前缀）

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4string s;
 5int nxt[200001],n;
 6
 7void kmp(){
 8    n=s.size();
 9    int j=0;
10    nxt[0]=0;
11    string t=s;
12    reverse(t.begin(),t.end());
13    s=s+"#"+t;
14    for(int i=1;i<n*2+1;i++){
15        while(j&&s[i]!=s[j])
16            j=nxt[j-1];
17        if(s[i]==s[j])
18            j++;
19        nxt[i]=j;
20    }
21    int ans=-1;
22    for(int i=n;i<2*n+1;i++){
23        ans=max(ans,nxt[i]);
24    }
25    for(int i=ans-1;i>=0;i--)cout<<s[i];
26}
27
28int main(){
29    cin>>s;
30    kmp();
31    return 0;
32}

exkmp#

洛谷 P5410

 1#include <bits/stdc++.h>
 2using namespace std;
 3const int N = 2e7+10;
 4typedef long long ll;
 5
 6string s,p;
 7ll z[N*2],n,m;
 8
 9void exkmp(string s,string p){
10    n=s.size();
11    m=p.size();
12    p=p+"#"+s;
13    ll L=0,R=-1;
14    z[0]=m;
15    for(int i=1;i<n+m+1;i++){
16        if(i>R)
17            z[i]=0;
18        else{
19            ll k=i-L;
20            z[i]=min(z[k],R-i);
21        }
22        while(i+z[i]<n+m+1&&p[z[i]]==p[i+z[i]])
23            ++z[i];
24        if(i+z[i]-1>R)
25            L=i,R=i+z[i]-1;
26    }
27    ll ans=0;
28    for(int i=0;i<m;i++)ans^=(1LL*(i+1)*(z[i]+1));
29    cout<<ans<<"\n";
30    ans=0;
31    for(int i=0;i<n;i++)ans^=(1LL*(i+1)*(z[i+m+1]+1));
32    cout<<ans<<"\n";
33}
34
35int main(){
36    cin>>s>>p;
37    exkmp(s,p);
38    return 0;
39}
