#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N = 2e5 + 7, mod = 1e9 + 7;

int n, q, u, v, op;
set<int> s[N];
int fa[N];

int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]);
}

void unite(int x, int y) {
    int fx = find(x), fy = find(y);
    if (fx != fy) {
        if (s[fx].size() < s[fy].size()) swap(fx, fy);
        fa[fy] = fx;
        for (int elem : s[fy]) {
            s[fx].insert(elem);
        }
        s[fy].clear();
    }
}

void solve() {
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        fa[i] = i;
        s[i].insert(i);
    }
    while (q--) {
        cin >> op >> u >> v;
        if (op == 1) {
            unite(u, v);
        } else {
            int x = find(u);
            if (s[x].size() < v) {
                cout << -1 << endl;
            } else {
                auto it = s[x].rbegin();
                advance(it, v - 1);
                cout << *it << endl;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    int t = 1;
    // cin >> t;
    while (t--) solve();
    return 0;
}
