ll qpow(ll x,ll k)
{
    ll ans=1,tmp=x;
    while(k) {
        if(k&1)  ans=ans*tmp%mod;
        tmp=tmp*tmp%mod;
        k>>=1;
    }

ll C(int x,int y)
{
    if(x<y)  return 0;

ll calc(int x,int y)
{
    if(x<y)  return 0;
