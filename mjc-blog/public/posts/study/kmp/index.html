KMP学习记录#

理念学习#

字符串匹配问题 查找s中p是否出现
暴力匹配 O(nm)
哈希优化 O(n+m)
KMP 线性复杂度解决 O(n+m)
KMP 中对于 s 中每个位置 i，我们要找到最大的 j 满足s[i - j +1]…s[i] 和 p[1]…[j] 相同
f[i] 对应j的位置，j对应p中的位置
如果 j 不等于 m，并且 s[i + 1] == p[j + 1]，j 右移一位
否则，j 向前回退，到满足 s[i - k + 1]…s[i] == p[1]…[j] 且 k 最大的位
如果 s[i + 1] 仍不等于 p[j + 1]，不停向前回退，直到相等或 j = 0

快速求出 k
易得到求 k 与 s 无关
最大的 k 满足 k < j,使得 p[1]..p[k] 和 p[j - k + 1]…p[j] 完全相同
可以用 next 维护每个 j 对应的 k

 1void kmp(){
 2    n=s.size()+1,m=p.size()+1;//字符串下标从 1 开始
 3    int j=0;
 4    nxt[1]=0;
 5    for(int i=2;i<=m;i++){
 6        while(j>0&&p[j+1]!=p[i])
 7            j=nxt[j];
 8        if(p[j+1]==p[i])
 9            j++;
10        nxt[i]=j;
11    }
12    j=0;
13    for(int i=1;i<=n;i++){
14        while((j==m)||(j>0&&p[j+1]!=s[i]))
15            j=nxt[j];
16        if(p[j+1]==s[i])
17            j++;
18        f[i]=j;
19    }
20}

P3375

 1#include <bits/stdc++.h>
 2using namespace std;
 3const int N = 1e6+10;
 4
 5int n, m;
 6string s, p;
 7int nxt[N], f[N];
 8
 9void kmp() {
10    n = s.size();
11    m = p.size();
12    nxt[0] = 0;
13    int j = 0;
14    for (int i = 1; i < m; i++) {
15        while (j > 0 && p[i] != p[j])
16            j = nxt[j - 1];
17        if (p[i] == p[j])
18            j++;
19        nxt[i] = j;
20    }
21    j = 0;
22    for (int i = 0; i < n; i++) {
23        while (j > 0 && s[i] != p[j])
24            j = nxt[j - 1];
25        if (s[i] == p[j])
26            j++;
27        f[i] = j;
28        if (j == m) {
29            cout << i - m + 2 << endl;
30            j = nxt[j - 1];
31        }
32    }
33    for (int i = 0; i < m; i++) {
34        cout << nxt[i] << " ";
35    }
36    cout << endl;
37}

EXKMP(Z-algorithm)#

理论学习#

线性时间复杂度求出字符串s和他的任意后缀 s[i]…s[n] 的最长公共前缀的长度 O(n)

与 kmp 的区别：一个是到 s[i] 结束，一个是从 s[i] 开始

定义 z[1] = 0，从 2 到 n 枚举 i，依次计算 z[i] 的值
计算 z[i] 时，前面的 z 都已经计算好了
对于 j，有 s[j]…s[j + z[j] - 1] 和 s[1]…s[z[j]] 完全相等

为了计算 z[i]，在枚举i的过程中，需要维护R的最大区间 [L, R],其中 L = j，R =j + z[i] - 1
初始时 L = 1,R = 0
如果 i <= R:
易知 s[L]…s[R] = s[1]…s[R - L + 1]
令 k = i - L + 1,i 与 k 的位置对应，此时s[i]..s[R] = s[k]..s[R - L + 1]
如果 z[k] < R - i + 1，说明从 k 开始匹配不到那么远，也就是从 i 开始匹配不到 R，此时 z[i] = z[k]
反之，说明可以匹配到 R 那么远，从 R+1 开始往后暴力
如果 i > R:
暴力枚举匹配，记得更新 L 和 R

 1void exkmp(){
 2    int L=1,R=0;
 3    z[1]=0;
 4    for(int i=2;i<=2;i++){
 5        if(i>R)
 6            z[i]=0;
 7        else{
 8            int k=i-L+1;
 9            z[i]=min(z[k],R-i+1);
10        }
11        while(i+z[i]<=n&&s[z[i]+1]==s[i+z[i]])
12            ++z[i];
13        if(i+z[i]-1>R)
14            L=i,R=i+z[i]=1;
15    }
16}

例题#

给出字符串 s, p,求 s 中 p 出现的次数和位置

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4string s,p;
 5int z[200001],n,m;
 6
 7void exkmp(){
 8    n=s.size();
 9    m=p.size();
10    p=p+"#"+s;
11    int L=0,R=-1;
12    z[0]=0;
13    for(int i=1;i<n+m+1;i++){
14        if(i>R)
15            z[i]=0;
16        else{
17            int k=i-L;
18            z[i]=min(z[k],R-i);
19        }
20        while(i+z[i]<n+m+1&&p[z[i]]==p[i+z[i]])
21            ++z[i];
22        if(i+z[i]-1>R)
23            L=i,R=i+z[i]-1;
24    }
25    int ans=0;
26    for(int i=m+1;i<m+n+1;i++)
27        if(z[i]==m)ans++;
28    cout<<ans<<endl;
29    for(int i=m+1;i<n+m+1;i++)
30        if(z[i]==m)cout<<i-m<<endl;
31}
