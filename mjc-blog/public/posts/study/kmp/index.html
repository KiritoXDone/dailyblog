const int N = 1e6+10;

int n, m;

int nxt[N], f[N];

void kmp() {
    n = s.size();
    m = p.size();
    nxt[0] = 0;
    int j = 0;
    for (int i = 1; i < m; i++) {
        while (j > 0 && p[i] != p[j])
            j = nxt[j - 1];
        if (p[i] == p[j])
            j++;
        nxt[i] = j;
    }
    j = 0;
    for (int i = 0; i < n; i++) {
        while (j > 0 && s[i] != p[j])
            j = nxt[j - 1];
        if (s[i] == p[j])
            j++;
        f[i] = j;
        if (j == m) {
            cout << i - m + 2 << endl;
            j = nxt[j - 1];
        }
    }
    for (int i = 0; i < m; i++) {
        cout << nxt[i] << " ";
    }
    cout << endl;
}

int main() {
    cin >> s >> p;
    kmp();
    return 0;
}
