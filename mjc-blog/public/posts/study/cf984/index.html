struct Node{
    ll id,val;
    bool operator<(const Node &a){return val>a.val;}
};

int n,k;
int b[N],c[N];
map<int,int>vis;

void solve(){
    cin>>k>>n;
    vis.clear();
    for(int i=1;i<=n;i++)cin>>b[i]>>c[i];
    for(int i=1;i<=n;i++){
        vis[b[i]]+=c[i];
    }
    vector<Node>v;
    for(auto [x,y]:vis){
        v.push_back(Node{x,y});
    }
    sort(v.begin(),v.end());
    ll ans=0;
    for(int i=0;i<min(k, (int)v.size());i++){
        ans+=v[i].val;
    }
    cout<<ans<<endl;
}

    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;

#include <bits/stdc++.h>
using namespace std;
#define ls now << 1
#define rs now << 1 | 1
#define endl "\n"
#define lowbit(x) ((x) & (-x))
typedef long long ll;
const int N = 1e5 + 7, mod = 1e9 + 7;

ll l,r,i,k;

ll fc(ll n){
    if(n%4==0) return n;
    if(n%4==1) return 1;
    if(n%4==2) return n+1;
    return 0;
}

ll fx(ll n,ll i,ll k){
    if(!i){
        if(!k)return fc(n);
        else return 0;
    }
    ll x=1ll<<i;
    if(n<k) return 0;
    ll t=(n-k)/x;
    ll cnt=t+1;
    ll res=fc(t)<<i;
    if(cnt%2)res^=k;
    return res;
}

void solve() {
    cin>>l>>r>>i>>k;
    ll s=fc(l-1)^fc(r);
    ll tmp=fx(r,i,k)^fx(l-1,i,k);
    ll ans=s^tmp;
    cout<<ans<<endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
