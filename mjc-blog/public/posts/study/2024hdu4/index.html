using namespace std;
#define ls now<<1
#define rs now<<1|1
typedef long long ll;
const int N=1e8+7, mod=1e9+7;

bool ip[N];
int n,k;
ll a[N];

int check(int lim){
    set<pair<int,int>> s;
    s.insert({0,0});
    int res=0;
    for(int i=1;i<=n;i++){
        auto it = s.begin();
        while(it != s.end()){
            if(a[i] - it->first < lim) it = s.end();
            else if(ip[i - it->second]) break;
            else it++;
        }
        if(it != s.end()) res++, s.clear();
        s.insert({a[i], i});
    }
    return res;
}

void solve(){
    cin >> n >> k;
    for(int i=1; i<=n; i++){
        cin >> a[i];
        a[i]+=a[i-1];
    }
    if(2*k > n){
        cout << "impossible\n";
        return;
    }
    int l = -2000, r = N;
    while(l < r){
        ll mid = (l + r + 1) >> 1;
        if(check(mid) >= k) l = mid;
        else r = mid - 1;
    }
    cout << l << endl;
}

int main(){

        }
    }
