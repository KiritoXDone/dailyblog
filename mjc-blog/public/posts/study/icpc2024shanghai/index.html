#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N=2e5+7;

struct Node{
	ll a,b;
	bool operator<(const Node &tmp){
		if(a==tmp.a)return b<tmp.b;
		return a<tmp.a;
	}
};

Node a[N];
ll c[N];
int cnt[N];
int n,fu,zheng;

bool check(ll x){
	int res=0;
	for(int i=1;i<=n;i++)cnt[i]=0;
	for(int i=1;i<=n;i++){
		if(a[i].a<0){
			ll l=1,r=fu,ans=0;
			while(l<=r){
				ll mid=(l+r)>>1;
				if(a[i].a*c[mid]+a[i].b>=x)l=mid+1,ans=mid;
				else r=mid-1;
			}
			cnt[ans]++;
		}
		else if(a[i].a==0){
			if(a[i].b>=x)res++;
		}
		else{
			ll l=zheng,r=n,ans=0;
			while(l<=r){
				ll mid=(l+r)>>1;
				if(a[i].a*c[mid]+a[i].b>=x)r=mid-1,ans=mid;
				else l=mid+1;
			}
			cnt[ans]++;
		}
	}
	int sum=0;
	for(int i=1;i<=fu;i++){
		sum++;
		if(cnt[i]){
			if(sum>=cnt[i])res=res+cnt[i],sum=sum-cnt[i];
			else res=res+sum,sum=0;
		}
	}
	sum=0;
	for(int i=n;i>=zheng;i--){
		sum++;
		if(cnt[i]){
			if(sum>=cnt[i])res+=cnt[i],sum-=cnt[i];
			else res+=sum,sum=0;
		}
	}
	return res>=(n+1)/2;
}

void solve(){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i].a;
	for(int i=1;i<=n;i++)cin>>a[i].b;
	for(int i=1;i<=n;i++)cin>>c[i];
	sort(a+1,a+1+n);
	sort(c+1,c+1+n);
	ll l=-2e18,r=2e18,ans=0;
	fu=0,zheng=n+1;
	for(int i=n;i>0;i--){
		if(a[i].a<0){
			fu=i;
			break;
		}
	}
	for(int i=1;i<=n;i++){
		if(a[i].a>0){
			zheng=i;
			break;
		}
	}
	while(l<=r){
		ll mid=ll(l+r)>>1;
		if(check(mid))l=mid+1,ans=mid;
		else r=mid-1;
	}
	cout<<ans<<endl;
}

int main(){
	int t;
	cin>>t;
	while(t--)solve();
	return 0;
}

	k++;
	vis[now]=1;
	for(int v:edges[now])edge[v].erase(now);
